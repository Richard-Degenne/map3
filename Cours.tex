\chapter{Transformée de Laplace}

\section{Introduction --- Définitions}

La résolution d'équations différentielles fait intervenir plusieurs opérations appliquées à $\R$, l'ensemble des réels. Ces opérations sont :

\begin{itemize}
	\item{L'addition ;}
	\item{La multiplication ;}
	\item{La différentiation.}
\end{itemize}

La transformée de Laplace transforme la différentiation en multiplication, mais modifie l'ensemble $\R$ à l'ensemble des polynômes à coefficients réels. On définit la transformation de Laplace, notée $\mathcal{L}$, par
\[f(t)\xrightarrow{\mathcal{L}} F(p) = \int_0^\infty f(t)\,e^{-pt}\diff{t}=\L{f(t)}\]
où $p$ représente le symbole de Laplace.

$F(p)$ est appelée \textbf{image} de $f(t)$. $f(t)$ est appelé \textbf{original} de $F(p)$, s'il existe. Il y a trois conditions à l'existence d'un original :

\begin{itemize}
	\item{$f(t)$ doit être causale, c'est-à-dire que $f(t)=0\forall t<0$.}
	\item{$f(t)$ doit être définie et continue sur $\R^+$, sauf en un nombre fini de points de discontinuité de 1\textsuperscript{ère} espèce.\footnote{Une discontinuité est dite de première espèce s'il existe une limite à droite et à gauche.}}
	\item{$f(t)$ doit être à croissance exponentielle, c'est-à-dire que $\exists c,M \mid\forall t>0,f(t)<M\,e^{ct}$}
\end{itemize}

On rappelle que $t$ est la variable du domaine temporel et $p=\sigma+j\omega$ une variable complexe. Ainsi, si $f(t)$ est majorée par $M\,e^{ct}$ et $\sigma>c$ alors l'exponentielle est négative et l'intégrale converge. On admettra dans la suite de ce cours que ces conditions sont toujours respectées.

\begin{example}
	\begin{align*}
	\L{1} & = \int_0^\infty1\,e^{-pt}\diff{t}\\
	      & = {\left[-\frac1p\,e^{-pt}\right]}_0^\infty\\
	      & = \frac1p
	\end{align*}
\end{example}
\begin{example}
	\begin{align*}
		\L{\cos t} & = \int_0^\infty\cos t\,e^{-pt}\diff{t}
	\end{align*}
	
	Ici, le calcul est tout de suite plus compliqué, on peut alors utiliser les propriétés de la transformation de Laplace et des tables de transformées. Ici, on pourrait transformer le cosinus en exponentielle et calculer la transformée de l'exponentielle
	\begin{align*}
	\L{e^t} & = \int_0^\infty e^t\,e^{-pt}\diff{t}\\
	        & = \int_0^\infty e^{-(p-1)t}\diff{t}\\
	        & = \frac1{p-1}
	\end{align*}
\end{example}

\section{Propriétés}
\subsection{Linéarité}
\[\L{a\,f(t)+b\,g(t)}=a\,\L{f(t)}+b\,\L{g(t)}\]

\subsection{Similitude}

\begin{align*}
	F(kp) & = \int_0^\infty e^{-pkt}\diff{t}\\
	      & = \int_0^\infty f\left(\frac\tau k\right)\,e^{-p\tau}\diff{\frac\tau k}\qquad(\tau=kt)\\
	      & = \frac1k\,\L{f\left(\frac\tau k\right)}
\end{align*}

De même, $\L{f(kt)}=\frac1k\,F\left(\frac pk\right)$.

\subsection{Translation}
\subsubsection{Du symbole de Laplace $p$}
\begin{align*}
	F(p+a) & = \int_0^\infty f(t)\,e^{-(p+a)t}\diff{t}\\
	       & = \int_0^\infty e^{-at}f(t)\,e^{-pt}\diff{t}\\
	       & = \L{e^{-at}f(t)}
\end{align*}

De même, $\L{e^{-at}f(t)}=F(p+a)$.

\subsubsection{De la variable temporelle $t$}

On apelle \emph{retard} toute valeur $\tau > 0$ telle que $f(t-\tau)=0\forall t<\tau$.

\begin{align*}
	\L{f(t-\tau)} & = \int_0^\infty f(t-\tau)\,e^{-pt}\diff{t}\\
	     \L{f(u)} & = \int_0^\infty f(u)\,e^{-p(u+\tau)}\diff{u}\qquad(u=t-\tau)\\
	              & = e^{-pt}\int_0^\infty f(u)\,e^{-pu}\diff{u}\\
	              & = e^{-pt}F(p)
\end{align*}

\begin{example}[Les fonctions périodiques]
	Soit $f(t)$ une fonction de période $T$. On va ici décomposer la fonction sur chaque période. On définit $f_0(t)=f(t)\forall t\in[0;T] ; 0\forall t\geq T$.
	\begin{figure}[h!]
	\begin{center}
		\begin{tikzpicture}
			\draw[->] (-0.1,0) -- (10,0) node[anchor=north west]{$t$};
			\draw[->] (0,-0.1) -- (0,4);
			\draw (4,2pt) -- (4,-2pt) node[anchor=north]{$T$};
			\draw (8,2pt) -- (8,-2pt) node[anchor=north]{$2T$};

			\draw[red, very thick] (0,2) sin (1,3) cos (2,2) sin (3,1) cos (4,2) sin (5,3) cos (6,2) sin (7,1) cos (8,2) sin (9,3) node[anchor=west,red]{$f(t)$};
			\draw[blue] (0,2) sin (1,3) cos (2,2) sin (3,1) cos (4,2) -- (4,0) -- (9,0) node[anchor=south west]{$f_0(t)$};
		\end{tikzpicture}
	\end{center}
	\caption{Décomposition d'une fonction périodique}
	\end{figure}
	\begin{align*}
		\L{f(t)} & = \L{f_0(t)  +f_0(t-T)      +f_0(t-2T)+\dotsb}\\
		         & = \L{f_0(t)} +\L{f_0(t-T)}  +\L{f_0(t-2T)}+\dotsb\\
			 & = F_0(p)     +e^{-pT}F_0(p) +e^{-2pT}F_0(p)+\dotsb\\
			 & = (1+e^{-pT}+e^{-2pT}+\dotsb)F_0(p)\\
			 & = \frac{1}{1-e^{-pT}}F_0(p)\\
	\end{align*}
	Pour les fonctions périodiques usuelles, il faut plutôt utiliser une table de transformée plutôt que cette méthode.
\end{example}

\subsection{Dérivation}
\subsubsection{De l'original}
\begin{align*}
	\L{\dot{f}(t)} & = \int_0^\infty\dot{f}(t)\,e^{-pt}\diff{t}\\
	               & = {\left[f(t)\,e^{-pt}\right]}_0^\infty - \int_0^\infty f(t)\times(-p)e^{-pt}\diff{t}\\
	               & = pF(p)-f(0^+)
\end{align*}

On a donc bien transformé la différentiation du domaine temporel en multiplication dans le domaine symbolique. De manière générale, on écrit
\[\L{f^{(n)}(t)} = p^nF(p)-p^{n-1}f(0^+)-p^{n-2}\dot{f}(0^+)-\dotsb-f^{(n-1)}(0^+)\]

\subsubsection{De l'image}
\begin{align*}
	\frac{\partial F}{\partial p} & = \frac\partial{\partial p}\left(\int_0^\infty f(t)\,e^{-pt}\diff{t}\right)\\
	                              & = \int_0^\infty f(t)\frac{\partial(e^{-pt})}{\partial p}\diff{t}\\
				      & = -\int_0^\infty t\,f(t)\,e^{-pt}\diff{t}\\
				      & = -\L{t\,f(t)}\\
\end{align*}

De même, $\L{t\,f(t)}=-\frac{\partial F}{\partial p}$.

\begin{example}
	\begin{align*}
		\L{t}   & = \L{1\times t} = -\frac\partial{\partial p}\left(\frac1p\right)=\frac1{p^2}\\
		\L{t^2} & = \L{t\times t} = \frac1{p^3}\\
		\L{t^n} & = \frac1{p^{n+1}}
	\end{align*}
\end{example}

\subsection{Comportement asymptotique}

Le but ici est de déterminer le comportement de $f(t)$ à partir de $F(p)$ sans avoir à effectuer la transformation inverse.

\begin{itemize}
	\item{Si $f(t)$ est une fonction, alors $\lim_{p\to\infty}=0$.}
	\item{Si $\int_0^\infty f(t)\diff{t}$ converge, alors $\lim_{p\to0}=\int_0^\infty f(t)\diff{t}$.}
\end{itemize}

\subsubsection{Théorèmes de la valeur initiale et de la valeur finale}

\begin{align*}
	\lim_{t\to0^+}f(t)    & = \lim_{p\to\infty}pF(p)\\
	\lim_{t\to\infty}f(t) & = \lim_{p\to0^+}pF(p)\qquad\text{(uniquement si }f(t)\text{ converge)}
\end{align*}

\subsection{Produit de convolution}

Le produit de convolution est une opération notée $*$ définie par
\[f(t)*g(t) = \int_0^\infty f(\tau)\,g(t-\tau)\diff{\tau}\]

\subsubsection{Dans le domaine temporel}

Le produit de convolution de deux fonctions temporelles est utilisé en automatique pour définir l'effet d'une fonction de transfert sur un signal d'entrée. Le passage dans le domaine de Laplace simplifie cette opération, puisqu'on a la propriété suivante :
\[\L{f(t)*g(t)} = F(p)\,G(p)\]

\subsubsection{Dans le domaine symbolique}

À l'inverse, si une fonction temporelle peut s'écrire comme le rpoduit de deux fonctions $f$ et $g$, alors
\[\L{f(t)\,g(t)} = F(p)*G(p)\]

\section{Inversion de la transformée}

Pour réaliser la transformée de Laplace inverse se calcule généralement à partir d'une table, mais il faut pour cela que la transformée de Laplace soit une application bijective, c'est-à-dire qu'à une fonction temporel ne corresponde qu'une unique image et qu'à une fonction symbolique ne corresponde qu'un unique original. On l'admettra ici, mais c'est le cas.

Soient $f_1(t)$ et $f_2(t)$ ayant respectivement pour transformée $F_1(p)$ et $F_2(p)$. Si on a
\[\exists p_0\mid\forall p\geq p_0,F_1(p)=F_2(p)\]
alors on a $f_1=f_2$, sauf peut-être aux points de discontinuité.

\begin{example}
	Nous savons déjà que $\L{\Gamma(t)}=\frac1p$. On peut dès lors affirmer
	\[\Li{\frac1p}=\left\{
	\begin{array}{ll}
		0      & \forall t<0\\
		\alpha & \text{arbitraire pour }t=0\\
		1      & \forall t>0
	\end{array}\right.\]
	On note que la transformée inverse ne permet pas de donner les valeurs de la fonction temporelle en ses points de discontinuité. Il faut les fixer arbitrairement.
\end{example}

La résolution d'une transformée inverse peut se faire à l'intégrale dite de Mellin-Fourier :
\[\Li{F(p)}=\frac1{2\pi j}\int_{c-j\infty}^{c+j\infty}F(p)\,e^{pt}\diff{p}\]

Il s'agit d'une intégrale à variable complexe dont le domaine d'intégration est délimité dans le plan complexe par la droite $\Re(z)=c$ et fermée à gauche de manière à contenir toutes les singularités\footnote{Par \emph{singularité}, on parle ici des pôles, racines du dénominateur de la fraction rationnelle que constitue $F(p)$.} de $F(p)$.

\subsection{Théorème des résidus}

Soit un point isolé $a$ de la fonction $F(p)$. On peut représenter $F(p)$ autour de $a$, dans un cercle de centre $a$ et de rayon arbitrairement petit, par sa série de Laurent. On appelle résidu de $F(p)$ en $a$ le coefficient $r$ de $(p-a)^{-1}$, noté $r_{-1}(a)$ du développement en série de Laurent de $F(p)$. Il a pour expression
\[r_{-1}(a) = \frac{1}{2\pi j}\int_\Gamma F(p)\diff{p}\]
où $\Gamma$ est une courbe fermée du plan complexe entourant la singularité $a$ (et uniquement cette singularité) et est parcourue dans le sens direct.

\paragraph{Théorème} Soit $F(p)$ une fonction uniforme et holomorphe\footnote{Terminologie équivalente de \emph{continue et dérivable} dans le domaine des fonctions complexes.} à l'intérieure et sur la frontière $\mathcal{C}$ d'un domaine $\mathcal{D}$, sauf en un nombre fini de points $a,b,\dotsc,l$ qui sont les pôles ou des points singuliers. On peut écrire
\[\int_\mathcal{C}F(p)\diff{p} = 2\pi j\big(r_{-1}(a) + r_{-1}(b) + \dotsb + r_{-1}(l)\big)\]

\begin{example}
\[\int_\text{cercle}e^{-2p}\diff{p}=0\]

En effet, la fonction ne présente aucun pôle.
\end{example}

On cherche, à partir de ce théorème, à calculer les résidus.

\begin{example}
	Soit $F(p) = G(p)\frac{1}{(p-a)^n}$. $a$ est un pôle d'ordre $n$ pour $F(p)$. On va donc calculer son résidu $r_{-1}(a)$. Pour alléger la notation, on notera dans les calculs suivants $r_{-1}$.
	
	\begin{align*}
	G(p) & = (p-a)^n\,F(p)\\
	F(p) & = r_{-n}(p-a)^{-n} + r_{-(n-1)}(p-a)^{-(n-1)} + \dotsb + r_{-1}(p-a)^{-1} + r_0 + r_1(p-a) + \dotsb\\
	G(p) & = r_{-n} + r_{-(n-1)}(p-a) + \dotsb + r_{-1}(p-a)^{n-1} + r_0(p-a)^n + \dotsb\\
	\end{align*}
	On dérive $n-1$ fois et on pose $p=a$.
	\[r_{-1} = \frac{1}{(n-1)!}{\left(\frac{\partial^{n-1}}{\partial\,p^{n-1}}(p-a)^n\,F(p)\right)}_{p=a}\]
\end{example}

Appliquons le théorème des résidus à la transformée de Laplace inverse. Pour cela, on reprend l'intégrale de Mellin-Fourier :
\begin{align*}
f(t) & = \frac{1}{2\pi j}\int_{c-j\infty}^{c+j\infty}F(p)\,e^{pt}\diff{p}\\
     & = \sum_\text{résidus} F(p)\,e^{pt}\\
\end{align*}

\begin{example}{$\Li{\frac{p+a}{p^2}}$}

Cette fonction ne présente qu'un seul pôle en $p=0$, d'ordre $2$.

\begin{align*}
r_{-1}(0) & = \frac{1}{(2-1)!}{\left(\frac{\partial}{\partial\,p}p^2\frac{p+a}{p^2}e^{pt}\right)}_{p=0}\\
	  & = \frac{1}{(2-1)!}{\left(e^{pt}+(p+a)t\,e^{pt}\right)}_{p=0}\\
	  & = 1+at
\end{align*}
Donc, $\Li{\frac{p+a}{p^2}}=1+at$.
\end{example}

\subsection{Développement en série de l'image}

On peut décomposer la fonction symbolique $F(p)$ en une série de fonction. On écrit
\[F(p)=\sum_{n\geq0}a_n\,F_n(p)\]

On peut alors retrouver la transformée de Laplace inverse :
\[f(t) = \sum_{n\geq0}a_n\,\Li{F_n(p)}\]

\begin{example}{$\Li{\frac{p}{p^2-a^2}}$}
\begin{align*}
F(p) & = \frac{p}{p^2\left(1-\frac{a^2}{p^2}\right)}\\
     & = \frac{1}{p}\left(1+\frac{a^2}{p^2}+\frac{a^4}{p^4}+\dotsb\right)\\
     & = \frac{1}{p}+\frac{a^2}{p^3}+\frac{a^4}{p^5}+\dotsb\\
f(t) & = 1+\frac{a^2t^2}{2!} + \frac{a^4t^4}{4!}+\dotsb\\
     & = \cosh(at)
\end{align*}
\end{example}

On utilise rarement cette méthode pour réaliser une transformation de Laplace inverse, mais on peut aisément retrouver des développements limités.

\chapter{Transformée en $\mathcal{Z}$}

Dans ce chapitre, on notera $\mathcal{Z}$ l'opération de transformation et $z$ la variable symbolique utilisée.

\section{Introduction}

La transformation de Laplace permet de traiter le cas des fonctions continues. Cependant, on peut être amené à traiter des varaibles discrètes, comme c'est le cas lorsque l'on se retrouve face à des équations de récurrence, définies par $x_{k} = f(x_{k-1}, x_{k-2},\dotsc)$. Résoudre ce genre d'équations revient à trouver l'ensemble $\big\{x_k\big\}$ des solutions.

\begin{example}
\[\left\{\begin{array}{l}
	x_{k+1} = A\,x_k + b_k\\
	k\geq0, x_0\text{ connu}
	\end{array}\right.\]

On commance par résoudre le système homogène, en négligeant le terme constant.
\begin{align*}
x_1 & = A\,x_0\\
x_2 & = A^2\,x_0\\
x_k & = A^k\,x_0
\end{align*}

On résout ensuite le système non-homogène :
\begin{align*}
x_1 & = A\,x_0+b_0\\
x_2 & = A^2\,x_0 + A\,b_0 + b_1\\
x_k & = A^k\,x_0 + \sum_{i=0}^{k-1}A^i\,b_{k-(1+i)}
\end{align*}
\end{example}

La transformée en $\mathcal{Z}$ a pour but de ne pas avoir à effectuer le calcul de la série.

\section{Définitions}

Soit une suite $\big\{x_k\big\}$ définie $\forall k\geq0$. On définit la série $\Z{x_k}$ par
\[\Z{x_k}=\sum_{k\geq0}x_k\,z^{-k}\]
avec $z\in\C$.

On suppose que la suite $\big\{x_k\big\}$ est à croissance exponentielle.

\begin{example}
\begin{align*}
\Z{1} = \sum_{k\geq0}1\times z^{-k} & = 1 + z^{-1} + z^{-2} + \dotsb\\
                               & = \frac{1}{1-z^{-1}}\\
			       & = \frac{z}{z-1}\\
\end{align*}
\end{example}

\begin{example}
\begin{align*}
\Z{e^{-kT}} & = \Z{e^{-t}}\qquad(t=kT,\text{on parle d'échantillonnage.})\\
            & = \sum_{k\geq0}e^{-kT}\,z^{-k}\\
	    & = \sum_{k\geq0}{(z\,e^{t})}^{-k}\\
	    & = \frac{z\,e^{T}}{z\,e^{T}-1}
\end{align*}
\end{example}

\begin{example}
\begin{align*}
\Z{k} & = \sum_{k\geq0}k\,z^{-k}\\
      & = 0 + z^{-1} + 2z^{-2} + \dotsb\\
\end{align*}

On ne connaît aucun développement limité de cette forme-ci. On utilise donc les propriétés et des tables de la transformée en $\mathcal{Z}$ pour les fonctions plus compliquées.
\end{example}

\section{Propriétés}

\subsection{Linéarité}

\[\Z{a\,x_k + b\,y_k} = a\,X_k(z) + b\,Y_k(z)\]

\subsection{Retard}

\begin{align*}
\Z{x_{k-1}} & = \sum_{k\geq0}x_{k-1}\,z^{-k}\\
            & = \sum_{k\geq0}x_{k-1}\,z^{-(k-1)} \times z^{-1}\\
	    & = z^{-1}\sum_{k\geq0}x^k\,z^{-k}\\
	    & = z^{-1}\,X_k(z)\\
\end{align*}

$z^{-1}$ est appelé opérateur de retard de la transformée en $\mathcal{Z}$. D'une manière générale, on peut écrire
\[\Z{x_{k-r}} = z^{-r}\,X_k(z)\]

\subsection{Avance}

Bien que la notion d'avance n'aie aucune réalité physique\footnote{Aucun évènement physique ne se produit avant sa cause. Par exemple, une pièce ne chauffera pas avant d'avoir allumé le chauffage.}, elle peut être particulièrement utile d'un point de vue purement mathématique.

\begin{align*}
\Z{x_{k+1}} & = \sum_{k\geq0}x_{k+1}\,z^{-k}\\
            & = \sum_{k\geq0}x_{k+1}\,z^{-(k+1)}\times z\\
            & = z\sum{k\geq1}x^k\,z^{-k}\\
            & = z\left(\sum_{k\geq0}x^k\,z^{-k}-x_0\right)\\
            & = z\,X_k(z)-z\,x_0
\end{align*}

Encore une fois, on peut généraliser cette formule à toute avance $r$.
\[\Z{x_{k+r}} = z^r\,X_k(z) - z^r\,x_0 - z^{r-1}\,x_1 - \dotsb - z\,x_{r-1}\]

\subsection{Comportement asymptotique}

\subsubsection{Théorème de la valeur initiale}

\begin{align*}
X_k(z) & = \sum_{k\geq0}x_k\,z^{-k}\\
       & = x_0 + x_1\,z^{-1} + x_2\,z^{-2} + \dotsb\\
\end{align*}

En posant $z=\infty$, tous les termes sauf le premier s'annulent. On en déduit
\[\lim_{z\to\infty}X_k(z) = x_0\]

\subsubsection{Théorème de la valeur finale}

On va poser ici $e_k = x_k - x_{k-1}$.

\begin{align*}
\Z{e_k} & = \Z{x_k - x_{k-1}}\\
        & = X_k(z) - z^{-1}\,X_k(z)\\
        & = (1-z^{-1})X_k(z)\\
\Z{e_k} & = \sum_{k\geq0}e_k\,z^{-k}\\
        & = \sum_{k\geq0}(x_k - x_{k-1})z^{-k}\qquad\text{(La somme se téléscope.)}
\end{align*}

D'où le théorème
\[\lim_{k\to\infty}x_k=\lim_{z\to1}(1-z^{-1})X_k(z)\]

\subsection{Intégration discrète}

On peut utiliser la transformée en $\mathcal{Z}$ pour traiter des séries. Soit $\big\{S_k\big\}$ définie par
\[S_k = \sum_{j=0}^{k}x_j\]

On va chercher ici à exprimer $\Z{S_k}$ en fonction de $\Z{x_k}$.
\begin{align*}
\Z{S_k}          & = \Z{\underbrace{\sum_{j=0}^{k-1}x_j}_{S_{k-1}} + x_k}\\
                 & = z^{-1}\,S_k(z) + X_k(z)\\
(1-z^{-1})S_k(z) & = X_k(z)\\
S_k(z)           & = \frac{1}{1-z^{-1}}X_k(z)
\end{align*}

Inversement, on a $X_k(z) = S_k(z) - z^{-1}\,S_k(z)$.
